DELIMITER //

CREATE PROCEDURE EliminarUsuario(_ID INT) 
BEGIN
    -- Actualizar las facturas relacionadas al usuario que se eliminará
    UPDATE Facturas SET ID_Usuario = 1 WHERE ID_Usuario = _ID;

    -- Eliminar al usuario
    DELETE FROM Usuario WHERE ID = _ID;
END //

DELIMITER ;
